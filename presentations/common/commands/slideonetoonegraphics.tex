\newcommand<>{\slideonetoonegraphics}[5]
{
	\slideonetoone{#1}
	{
		\begin{figure}
			\includegraphics[width=\textwidth, height=0.7\textheight, keepaspectratio]{#2}
			\caption{#3}
		\end{figure}	
	}
	{
		\begin{figure}
			\includegraphics[width=\textwidth, height=0.7\textheight, keepaspectratio]{#4}
			\caption{#5}
		\end{figure}	
	}
}
